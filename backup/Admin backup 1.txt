<script>
    let globalNotices = [];
    let globalEvents = [];

    document.addEventListener("DOMContentLoaded", fetchSidebarData);

    // Fetch live content and render cards on the right sidebar
    function fetchSidebarData() {
        fetch('/api/data')
            .then(res => res.json())
            .then(data => {
                globalNotices = data.notices || [];
                globalEvents = data.events || [];
                renderSidebar();
            });
    }

    function renderSidebar() {
        const noticeContainer = document.getElementById("sidebar-notices");
        const eventContainer = document.getElementById("sidebar-events");

        // Render Notices List
        if (globalNotices.length === 0) {
            noticeContainer.innerHTML = "<p style='color:#64748b; font-size:13px;'>No active notices.</p>";
        } else {
            noticeContainer.innerHTML = "";
            globalNotices.forEach(n => {
                noticeContainer.innerHTML += `
                    <div class="list-card">
                        <div class="card-details">
                            <strong>${n.title}</strong>
                            <small>${n.date}</small>
                        </div>
                        <div class="action-box">
                            <button class="btn-action btn-edit" onclick="setupEdit('notice', ${n.id})">Edit</button>
                            <button class="btn-action btn-del" onclick="deleteItem('notice', ${n.id})">Delete</button>
                        </div>
                    </div>
                `;
            });
        }

        // Render Events List
        if (globalEvents.length === 0) {
            eventContainer.innerHTML = "<p style='color:#64748b; font-size:13px;'>No active albums.</p>";
        } else {
            eventContainer.innerHTML = "";
            globalEvents.forEach(e => {
                eventContainer.innerHTML += `
                    <div class="list-card event-type">
                        <div class="card-details">
                            <strong>🎉 ${e.title}</strong>
                            <small>${e.images.length} Photos</small>
                        </div>
                        <div class="action-box">
                            <button class="btn-action btn-edit" onclick="setupEdit('event', ${e.id})">Edit</button>
                            <button class="btn-action btn-del" onclick="deleteItem('event', ${e.id})">Delete</button>
                        </div>
                    </div>
                `;
            });
        }
    }

    // Step 2 logic: Pull values from right side card and auto-fill inside left form
    function setupEdit(type, id) {
        if (type === 'notice') {
            const item = globalNotices.find(n => n.id === id);
            if (!item) return;
            document.getElementById("notice-id").value = item.id;
            document.getElementById("notice-title").value = item.title;
            document.getElementById("notice-desc").value = item.description;
            document.getElementById("notice-form-title").innerText = "✏️ Editing Notice: " + item.title;
            document.getElementById("notice-btn").innerText = "Save & Update Notice";
            document.getElementById("notice-btn").style.background = "#eab308";
            document.getElementById("notice-btn").style.color = "black";
            document.getElementById("notice-title").focus();
        } else if (type === 'event') {
            const item = globalEvents.find(e => e.id === id);
            if (!item) return;
            document.getElementById("event-id").value = item.id;
            document.getElementById("event-title").value = item.title;
            document.getElementById("event-desc").value = item.description;
            document.getElementById("event-form-title").innerText = "✏️ Editing Album Details: " + item.title;
            document.getElementById("event-btn").innerText = "Save & Update Album";
            document.getElementById("event-btn").style.background = "#eab308";
            document.getElementById("event-btn").style.color = "black";
            document.getElementById("event-photos").required = false; // Photos selection optional during update
            document.getElementById("photo-helper-text").innerText = "⚠️ Khali chhodne par purani photos safe rahengi. Naye chashme se add hongi.";
            document.getElementById("event-title").focus();
        }
    }

    // Delete item request handler
    function deleteItem(type, id) {
        if (confirm(`Kya aap sach me is ${type} ko permanently delete karna chahte hain?`)) {
            fetch(`/api/admin/delete/${type}/${id}`, { method: 'DELETE' })
                .then(res => res.json())
                .then(resData => {
                    alert(resData.message);
                    fetchSidebarData(); // Instantly update sidebar without full page reload
                })
                .catch(err => alert("Error deleting item"));
        }
    }
</script>
