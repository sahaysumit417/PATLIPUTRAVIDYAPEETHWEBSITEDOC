<script>
    function openEnquiryModal() {
        document.getElementById("enquiry-modal").style.display = "flex";
    }
    function closeEnquiryModal() {
        document.getElementById("enquiry-modal").style.display = "none";
    }

    // 🎯 ऑटोमैटिक लाइव एक्टिव नेविगेशन लिंकर इंजन
    document.addEventListener("DOMContentLoaded", () => {
        setTimeout(() => {
            const currentPath = window.location.pathname;
            const navLinks = document.querySelectorAll(".nav-item-link");
            
            navLinks.forEach(link => {
                const href = link.getAttribute("href");
                
                // चेक करें कि यूआरएल पाथ मैच हो रहा है या नहीं
                if (currentPath === href || (currentPath === "/" && href === "/")) {
                    link.style.setProperty('color', 'var(--accent-lime)', 'important');
                    link.style.setProperty('font-weight', '600', 'important');
                    
                    // इनलाइन एक्टिव बॉटम बार बनाना
                    const line = document.createElement("div");
                    line.style.cssText = "position: absolute; bottom: 0; left: 0; width: 100%; height: 2px; background: var(--accent-lime);";
                    link.appendChild(line);
                }
            });
        }, 120);
    });
</script>
