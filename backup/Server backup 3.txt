const express = require('express');
const path = require('path');
const fs = require('fs');
const multer = require('multer');

const app = express();
const PORT = 3000;

app.use(express.static(path.resolve(__dirname, 'public')));
app.use(express.json());
app.use(express.urlencoded({ extended: true }));

const DATA_FILE = path.resolve(__dirname, 'data', 'database.json');
const UPLOADS_DIR = path.resolve(__dirname, 'public', 'uploads');

if (!fs.existsSync(UPLOADS_DIR)) {
    fs.mkdirSync(UPLOADS_DIR, { recursive: true });
}

// --- MULTER STORAGE ---
const storage = multer.diskStorage({
    destination: (req, file, cb) => { cb(null, UPLOADS_DIR); },
    filename: (req, file, cb) => { cb(null, Date.now() + '-' + Math.round(Math.random() * 1E9) + path.extname(file.originalname)); }
});
const upload = multer({ storage: storage });

// --- VIEW ROUTES ---
app.get('/', (req, res) => { res.sendFile(path.resolve(__dirname, 'views', 'index.html')); });
app.get('/admin', (req, res) => { res.sendFile(path.resolve(__dirname, 'views', 'admin.html')); });

app.get('/api/data', (req, res) => {
    if (fs.existsSync(DATA_FILE)) {
        res.json(JSON.parse(fs.readFileSync(DATA_FILE, 'utf8')));
    } else {
        res.json({ notices: [], events: [] });
    }
});

// --- SMART POST ROUTES (CREATES OR UPDATES DYNAMICALLY) ---

// 1. Handle Notice Publish OR Edit Update
app.post('/api/admin/upload-notice', (req, res) => {
    const { noticeId, title, description } = req.body;
    let localData = { notices: [], events: [] };
    if (fs.existsSync(DATA_FILE)) { localData = JSON.parse(fs.readFileSync(DATA_FILE, 'utf8')); }

    if (noticeId) {
        // --- EDIT MODE ---
        let existingNotice = localData.notices.find(n => n.id === parseInt(noticeId));
        if (existingNotice) {
            existingNotice.title = title;
            existingNotice.description = description;
            existingNotice.date = new Date().toLocaleDateString('en-GB') + ' (Updated)';
        }
    } else {
        // --- CREATE MODE ---
        localData.notices.push({
            id: Date.now(),
            title,
            description,
            date: new Date().toLocaleDateString('en-GB')
        });
    }

    fs.writeFileSync(DATA_FILE, JSON.stringify(localData, null, 2));
    res.send('<script>alert("Notice Saved Successfully!"); window.location.href="/admin";</script>');
});

// 2. Handle Event Cover/Album Create OR Edit Update
app.post('/api/admin/upload-event', upload.array('eventPhotos', 15), (req, res) => {
    const { eventId, eventTitle, eventDescription } = req.body;
    const uploadedFiles = req.files ? req.files.map(f => `/uploads/${f.filename}`) : [];

    let localData = { notices: [], events: [] };
    if (fs.existsSync(DATA_FILE)) { localData = JSON.parse(fs.readFileSync(DATA_FILE, 'utf8')); }

    if (eventId) {
        // --- EDIT MODE ---
        let existingEvent = localData.events.find(e => e.id === parseInt(eventId));
        if (existingEvent) {
            existingEvent.title = eventTitle.trim();
            existingEvent.description = eventDescription;
            // Agar naye photos select kiye hain toh hi array me push honge, varna purane safe rahenge
            if (uploadedFiles.length > 0) {
                existingEvent.images = existingEvent.images.concat(uploadedFiles);
                existingEvent.coverImage = uploadedFiles[0]; // updates cover image to the latest upload
            }
        }
    } else {
        // --- CREATE MODE ---
        localData.events.push({
            id: Date.now(),
            title: eventTitle.trim(),
            description: eventDescription,
            coverImage: uploadedFiles[0] || '/uploads/default-event.jpg',
            images: uploadedFiles
        });
    }

    fs.writeFileSync(DATA_FILE, JSON.stringify(localData, null, 2));
    res.send('<script>alert("Gallery Data Updated Successfully!"); window.location.href="/admin";</script>');
});

// --- DYNAMIC DELETE ROUTE MANAGER (WITH AUTOMATIC PHYSICAL FILE CLEANUP) ---
app.delete('/api/admin/delete/:type/:id', (req, res) => {
    const { type, id } = req.params;
    if (!fs.existsSync(DATA_FILE)) return res.status(404).json({ message: "Database not found" });

    let localData = JSON.parse(fs.readFileSync(DATA_FILE, 'utf8'));
    const itemId = parseInt(id);

    if (type === 'notice') {
        localData.notices = localData.notices.filter(n => n.id !== itemId);
    } else if (type === 'event') {
        // 1. Pehle us event album को ढूंढें जिसे डिलीट करना है
        const eventToDelete = localData.events.find(e => e.id === itemId);
        
        if (eventToDelete && eventToDelete.images) {
            // 2. Loop चलाकर उस एल्बम की सारी फोटो फाइलों को कंप्यूटर हार्डड्राइव से डिलीट करें
            eventToDelete.images.forEach(imgUrl => {
                const filename = path.basename(imgUrl); // फ़ाइल का नाम निकालेगा
                const physicalPath = path.join(UPLOADS_DIR, filename); // फ़ोल्डर का पूरा रास्ता

                try {
                    if (fs.existsSync(physicalPath)) {
                        fs.unlinkSync(physicalPath); // कंप्यूटर से हमेशा के लिए डिलीट
                        console.log(`Successfully Removed File: ${filename}`);
                    }
                } catch (err) {
                    console.error(`Error removing file: ${filename}`, err);
                }
            });
        }
        
        // 3. फ़ोल्डर साफ करने के बाद डेटाबेस से भी एल्बम हटा दें
        localData.events = localData.events.filter(e => e.id !== itemId);
    } else {
        return res.status(400).json({ message: "Invalid type requested" });
    }

    fs.writeFileSync(DATA_FILE, JSON.stringify(localData, null, 2));
    res.json({ message: `Successfully deleted the requested ${type} and cleared its files!` });
});

app.listen(PORT, () => { console.log(`Server running at http://localhost:${PORT}`); });
